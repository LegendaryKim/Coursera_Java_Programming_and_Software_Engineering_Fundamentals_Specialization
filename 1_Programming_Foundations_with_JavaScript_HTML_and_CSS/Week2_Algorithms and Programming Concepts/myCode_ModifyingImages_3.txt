function BluetoYello(image) {
    for (var pixel of image.values()) {
        var red = pixel.getRed();
        var green = pixel.getGreen();
        var blue = pixel.getBlue();
        if (red < 255 & green < 255 & blue < 255) {
            pixel.setRed(255);
            pixel.setGreen(255);
            pixel.setBlue(0);
        }
    }
    return image;
}

print(img);
