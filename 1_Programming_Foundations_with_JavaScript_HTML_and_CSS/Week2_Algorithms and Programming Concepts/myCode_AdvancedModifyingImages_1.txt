var frImg = new SimpleImage("drewRobert.png");
var bcImg = new SimpleImage("dinos.png");

var output = new SimpleImage(frImg.getWidth(), frImg.getHeight());

for (var pixel of frImg.values()) {
    var x = pixel.getX();
    var y = pixel.getY();
    if (pixel.getGreen() > pixel.getRed() + pixel.getBlue()) {
        output.setPixel(x,y, bcImg.getPixel(x,y));
    }
    else {
        output.setPixel(x,y, pixel);
    }
}
print(output)
